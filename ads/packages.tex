% Images
\usepackage{graphicx}
\graphicspath{ {./images/} }
\usepackage{wrapfig}
% TikZ and TikZ-libraries ("Drawing"-program)
\usepackage{tikz}
\usetikzlibrary{shapes.geometric}
\usetikzlibrary{automata, positioning, arrows}

% Multi-line tables
\usepackage{makecell}
\renewcommand{\cellalign} {cl} % Default cell alignment: Vertical centered and horizontally left-handed

% Source code
\usepackage{listings}
\lstset{ % https://en.wikibooks.org/wiki/LaTeX/Source_Code_Listings
  language = bash,            % the language of the code
  basicstyle = \ttfamily,
  breakatwhitespace = false,  % sets if automatic breaks should only happen at whitespace
  breaklines = true,          % sets automatic line breaking
  captionpos=b,               % sets the caption-position to bottom
  escapeinside = {\%(*} {*)}, % if you want to add LaTeX within your code
  frame = single,             % adds a frame around the code
  commentstyle = \color{lightGreen}, % comment style
  numberstyle = \scriptsize,
  xleftmargin = 16pt,         % Versetzt Listing, dass Zahlen nicht im Rand sind
  numbers = left,
  stepnumber = 1,
  numbersep = 8pt,
  % stringstyle = \color{darkRed},
  showstringspaces = false,
}

% Line spacing
\usepackage{setspace}

% Line and paragraph spacing (Absatzformatierung)
\setlength{\parindent} {0em}
\setlength{\parskip} {1em}

% Chapter formatting
\titleformat{\chapter} {\bfseries\LARGE} {\thechapter} {20pt} {\bfseries\LARGE}

% Spacing around section headings
\titlespacing\chapter {0pt} {-24pt plus 4pt minus 2pt} {20pt plus 2pt minus 2pt}
\titlespacing\section {0pt} {24pt plus 4pt minus 2pt} {20pt plus 2pt minus 2pt}
\titlespacing\subsection {0pt} {12pt plus 4pt minus 2pt} {8pt plus 2pt minus 2pt}
\titlespacing\subsubsection {0pt} {8pt plus 4pt minus 2pt} {8pt plus 2pt minus 2pt}
\makeatletter
\renewcommand\paragraph{\@startsection{paragraph} {4} {\z@}%
  {8pt plus 4pt minus 2pt}%
  {8pt plus 2pt minus 2pt}%
  {\normalfont\normalsize\bfseries}}
\makeatother
